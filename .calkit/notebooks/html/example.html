def f(k, c, n):
    return c*k**n

from scipy.optimize import curve_fit
krange = np.where(np.logical_and(m.k[0, :, 0] >= 2e-4, m.k[0, :, 0] <= 1e-3))
popt, pcov = curve_fit(f, m.k[0, krange, 0][0], ke[krange][:, 0], p0=[1e-9, -3])
print('c =', popt[0])
print('n =', popt[1])

fig, ax = plt.subplots(1)
